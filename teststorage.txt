int tmp[650000];

int ack(int m,int n){
    int res;
    if(tmp[m*70000+n]!=-1){
        return (tmp[m*70000+n]);
    }
	if(m==0){
		res= (n+1);
	}
	else if(n==0){
		res= (ack(m-1,1));
	}
	else{
        res= (ack(m-1,ack(m,n-1)));
	}
    tmp[m*70000+n]=res;
    return (res);
}

void main(){
	int i,j;
	for(i=0;i<10;i=i+1){
		for(j=0;j<70000;j=j+1){
			tmp[i*70000+j]=-1;
		}
	}
	printf(ack(4,1));

}
